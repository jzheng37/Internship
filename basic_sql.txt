CREATE DATABASE school_db;
USE school_db;

CREATE TABLE students (
    id INT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    height DECIMAL(5, 2)
);

SET height = 185.0 
WHERE id = 1;

//delete
DELETE FROM students
